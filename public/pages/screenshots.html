Query Examples

Get Latest 10 Calculations

SELECT * FROM calculations 
ORDER BY timestamp DESC 
LIMIT 10;

Filter by Currency

SELECT * FROM calculations 
WHERE currency = 'INR' 
ORDER BY timestamp DESC;

Statistics

SELECT 
  currency,
  COUNT(*) as total_calculations,
  AVG(amount) as avg_amount,
  SUM(amount) as total_amount
FROM calculations
GROUP BY currency;
